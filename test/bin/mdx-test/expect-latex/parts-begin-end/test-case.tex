Mdx can also understand ocaml code blocks:


% $MDX file=parts-begin-end.ml,part=toto
\begin{ocaml}
# let x = 3;;
val x : int = 3
# let y = 4;;
val y : int = 4
# print_int x;;
3
- : unit = ()
# print_int y;;
4
- : unit = ()
\end{ocaml}
